module Graph2Tree;

create OUT : Tree from IN : Graph;

rule Node2Node {
	from
		s: Graph!Node
	to
		t: Tree!Node
		(
			children <- s.children->collect(e | thisModule.child2child(e))
		)
}

lazy rule child2child {
	from 
		s: Graph!Node
	to
		t: Tree!Node
		(
			children <- s.children->collect(e | thisModule.child2child(e))
		)
}
